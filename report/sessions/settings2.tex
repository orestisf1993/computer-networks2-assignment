\newcommand{\plothere}[2]{%
\begin{figure}[htb]%
\centering%
\includegraphics[keepaspectratio, width=1.0\linewidth]{\plotfolder/#1}%

\label{fig:#1}%
\end{figure}%
}
\newcommand{\dplothere}[3]{%
\begin{figure}[htb]
\centering
\begin{subfigure}{\linewidth}
\centering
    \includegraphics[height=0.48\textheight]{\plotfolder/#1}
\end{subfigure}
\par\bigskip % force a bit of vertical whitespace
\begin{subfigure}{\linewidth}
\centering
    \includegraphics[height=0.48\textheight]{\plotfolder/#2}
\end{subfigure}
\caption{#3}
\label{fig:#1#2}%
\end{figure}
}
